function handleRegister(e) { e.preventDefault(); // 按下按鈕後立刻改變文字，避免客人重複點擊 const btnSubmit = document.getElementById('btnSubmit'); const originalBtnText = btnSubmit.innerText; btnSubmit.innerText = "處理中..."; btnSubmit.disabled = true; const name = document.getElementById('txtName').value.trim(); const phone = document.getElementById('txtPhone').value.trim(); const email = document.getElementById('txtEmail').value.trim(); const formattedResult = `【新客註冊資料】\n姓名：${name}\n電話：${phone}\nEmail：${email}`; // 嚴格檢查：是否在 LINE 客戶端內 if (!liff.isInClient()) { alert("❌ 請回到 LINE 聊天室，點擊「最新」的卡片進行註冊！\n(您目前可能點到了舊連結或使用外部瀏覽器)"); btnSubmit.innerText = originalBtnText; btnSubmit.disabled = false; return; } // 發送訊息 liff.sendMessages([{ type: 'text', text: formattedResult }]).then(() => { // 發送成功後，加一個提示讓客人安心，然後關閉視窗 alert("🎉 註冊成功！系統將為您自動確認預約。"); liff.closeWindow(); }).catch((err) => { console.error("Error sending message: ", err); // 錯誤攔截：精準告訴您哪裡出錯 alert("❌ 發送失敗！\n請檢查 LINE Developers 後台是否已勾選「chat_message.write」權限。\n錯誤碼: " + err.message); btnSubmit.innerText = originalBtnText; btnSubmit.disabled = false; }); }
